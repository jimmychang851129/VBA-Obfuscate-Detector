Function KTS16(NameSh As String) As Boolean
  On Error Resume Next

End Function
Function KT_WbOP16(WBName As String) As Boolean
    Dim WB As Workbook
    On Error Resume Next
    Set WB = Workbooks(WBName)
    If Err = 0 Then KT_WbOP16 = True Else KT_WbOP16 = False
End Function
